Feature: Couriers (my orders & delivered)

  Background:
    * url serviceHostUrl
    * header Accept = '*/*'
    * header Content-Type = 'application/json'
    # Login како courier
    * def courierCreds = { username: 'courier', password: 'courier' }
    Given path '/api/user/login'
    And request { username: '#(courierCreds.username)', password: '#(courierCreds.password)' }
    When method post
    Then status 200
    * def TOKEN = response.token
    * header Authorization = 'Bearer ' + TOKEN

  @my_orders
  Scenario: Courier - my assigned orders
    Given path '/api/couriers/my-orders'
    When method get
    Then status 200
    And match response == '#[]'

  @my_delivered
  Scenario: Courier - my delivered orders
    Given path '/api/couriers/my-delivered-orders'
    When method get
    Then status 200
    And match response == '#[]'
